*** 2.1.3 验证 C++ 环境

创建测试项目:

#+begin_src sh
mkdir -p ~/qt-test && cd ~/qt-test

# 创建 CMakeLists.txt
cat > CMakeLists.txt << 'EOF'
cmake_minimum_required(VERSION 3.16)
project(HelloQt LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)

find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets)

add_executable(hello main.cpp)
target_link_libraries(hello Qt6::Core Qt6::Gui Qt6::Widgets)
EOF

# 创建 main.cpp
cat > main.cpp << 'EOF'
#include <QApplication>
#include <QPushButton>
#include <QMessageBox>

int main(int argc, char *argv[])
{
    QApplication app(argc, argv);

    QPushButton button("Click Me!");
    QObject::connect(&button, &QPushButton::clicked, [&]() {
        QMessageBox::information(nullptr, "Hello", "Qt6 is working!");
    });
    button.show();

    return app.exec();
}
EOF

# 构建并运行
cmake -S . -B build
cmake --build build
./build/hello
#+end_src


*** 2.1.4 CMake 编译 Qt 程序详解

Qt 程序的编译过程与传统 C++ 程序不同, 它包含一个**预编译阶段**. 理解这个过程对于调试和优化构建非常重要.

**** Qt 预编译工具

Qt 提供了三个核心预编译工具:

| 工具 | 全称                 | 输入                   | 输出      | 作用                                         |
|------+----------------------+------------------------+-----------+----------------------------------------------|
| *moc*  | Meta-Object Compiler | 含 ~Q_OBJECT~ 宏的头文件 | ~moc_*.cpp~ | 处理信号/槽, 属性系统, 运行时类型信息        |
| *uic*  | UI Compiler          | ~.ui~ 文件               | ~ui_*.h~    | 将 Qt Designer 的 UI 文件转换为 C++ 头文件   |
| *rcc*  | Resource Compiler    | ~.qrc~ 文件              | ~qrc_*.cpp~ | 将资源文件(图片, 翻译文件等)编译为二进制代码 |

**** CMake 自动处理

现代 CMake 通过以下变量自动调用 Qt 预编译工具:

#+begin_src cmake
# 必须设置的语言标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 自动运行 moc (处理 Q_OBJECT 宏)
set(CMAKE_AUTOMOC ON)

# 自动运行 uic (处理 .ui 文件)
set(CMAKE_AUTOUIC ON)

# 自动运行 rcc (处理 .qrc 资源文件)
set(CMAKE_AUTORCC ON)
#+end_src

**** 完整示例

下面是一个包含 UI 文件和资源文件的完整项目结构:

#+begin_src
myapp/
├── CMakeLists.txt
├── main.cpp
├── mainwindow.cpp
├── mainwindow.h
├── mainwindow.ui      # Qt Designer UI 文件
└── resources.qrc      # 资源文件清单
#+end_src

***** CMakeLists.txt

#+begin_src cmake
cmake_minimum_required(VERSION 3.16)
project(MyApp LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 启用 Qt 预编译工具
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets)

add_executable(myapp
    main.cpp
    mainwindow.cpp
    mainwindow.h
    mainwindow.ui        # CMake 自动调用 uic 生成 ui_mainwindow.h
    resources.qrc        # CMake 自动调用 rcc 生成 qrc_resources.cpp
)

target_link_libraries(myapp Qt6::Core Qt6::Gui Qt6::Widgets)
#+end_src

***** resources.qrc (资源文件)

#+begin_src xml
<RCC>
    <qresource prefix="/">
        <file>images/icon.png</file>
        <file>styles/app.qss</file>
    </qresource>
</RCC>
#+end_src

***** mainwindow.h (需要 moc 处理)

#+begin_src cpp
#ifndef MAINWINDOW_H
#define MAINWINDOW_H

#include <QMainWindow>

// Q_OBJECT 宏指示 moc 需要为这个类生成元对象代码
class MainWindow : public QMainWindow {
    Q_OBJECT  // <-- moc 会读取这个宏

public:
    explicit MainWindow(QWidget *parent = nullptr);

signals:
    void userClicked();  // 信号声明

private slots:
    void onButtonClick();  // 槽函数声明
};

#endif
#+end_src

**** 手动运行预编译工具(调试用)

虽然 CMake 会自动处理, 但在调试时你可能需要手动运行:

#+begin_src sh
# 手动运行 moc
moc mainwindow.h -o moc_mainwindow.cpp

# 手动运行 uic
uic mainwindow.ui -o ui_mainwindow.h

# 手动运行 rcc
rcc resources.qrc -o qrc_resources.cpp
#+end_src

**** 编译流程图

完整的 Qt 程序编译流程:

#+begin_example
mainwindow.h --(moc)--> moc_mainwindow.cpp --+---> 编译器 ---> 链接器 ---> 可执行文件
                                            |
mainwindow.ui --(uic)--> ui_mainwindow.h ----+---> 编译器 ---> 链接器 ---> 可执行文件
                                            |
resources.qrc --(rcc)--> qrc_resources.cpp -+---> 编译器 ---> 链接器 ---> 可执行文件
                                            |
main.cpp -----------------------------------+---> 编译器 ---> 链接器 ---> 可执行文件
                                            |
mainwindow.cpp ----------------------------+---> 编译器 ---> 链接器 ---> 可执行文件
#+end_example

**** 常见问题

1. *"undefined reference to vtable"* 错误
   - 原因: CMake 没有找到 ~Q_OBJECT~ 宏, 未运行 moc
   - 解决: 确保设置了 ~CMAKE_AUTOMOC ON~

2. *"ui_xxx.h: No such file or directory"* 错误
   - 原因: uic 未运行或输出目录不正确
   - 解决: 确保设置了 ~CMAKE_AUTOUIC ON~

3. *资源文件无法加载*
   - 原因: rcc 未运行或 qrc 文件路径错误
   - 解决: 确保设置了 ~CMAKE_AUTORCC ON~, 并检查 qrc 中的文件路径

#+begin_src sh
pip install PySide6
#+end_src

可选依赖:

#+begin_src sh

*** 2.2.4 验证 Python 环境

创建测试脚本:

#+begin_src sh
cat > test_qt.py << 'EOF'
from PySide6.QtWidgets import QApplication, QPushButton, QMessageBox

def on_click():
    QMessageBox.information(None, "Hello", "PySide6 is working!")

app = QApplication([])

button = QPushButton("Click Me!")
button.clicked.connect(on_click)
button.show()

app.exec()
EOF

python test_qt.py
#+end_src

*** 2.3.3 配置项目依赖

创建项目目录和 =deps.edn=:

#+begin_src sh
mkdir -p ~/qt6-clojure && cd ~/qt6-clojure

cat > deps.edn << 'EOF'
{:paths ["src"]
 :deps {org.clojure/clojure {:mvn/version "1.11.1"}
        clj-python/libpython-clj {:mvn/version "2.025"}}
 :aliases {:run {:main-opts ["-m" "hello-qt"]}}}
EOF

mkdir -p src
#+end_src

*** 2.3.4 配置 Python 桥接

Clojure 通过 libpython-clj 调用 PySide6, 需要配置 Python 路径:

#+begin_src clojure
;; src/hello_qt.clj
(ns hello-qt
  (:require [libpython-clj2.python :as py]
            [libpython-clj2.require :refer [require-python]]))

(defn -main []
  ;; 初始化 Python
  (py/initialize!)

  ;; 导入 PySide6
  (require-python '[PySide6.QtWidgets :as widgets])

  (let [app (widgets/QApplication (py/list []))
        button (widgets/QPushButton "Click Me!")]
    (py/call-attr button "show")
    (py/call-attr app "exec")))
#+end_src

*** 2.3.5 验证 Clojure 环境

#+begin_src sh
clojure -M:run
#+end_src


** 2.4 Basilisp 环境搭建

*** 2.4.1 安装 Python 依赖

Basilisp 基于 Python, 需要先安装:

#+begin_src sh
mkdir -p ~/qt6-basilisp && cd ~/qt6-basilisp

# 创建虚拟环境
python3 -m venv venv
source venv/bin/activate

# 安装 Basilisp 和 PySide6
pip install basilisp PySide6
#+end_src

*** 2.4.2 配置 Lisp 入口点

#+begin_src sh
mkdir -p src

cat > src/hello_qt.lpy << 'EOF'
(ns hello-qt
  (:import [PySide6.QtWidgets QApplication QPushButton QMessageBox]))

(defn on-click []
  (QMessageBox/information nil "Hello" "Basilisp + Qt6 is working!"))

(defn -main []
  (let [app (QApplication [])
        button (QPushButton "Click Me!")]
    (.connect (.clicked button) on-click)
    (.show button)
    (.exec app)))

(-main)
EOF
#+end_src

*** 2.4.3 运行 Basilisp 程序

#+begin_src sh
basilisp run src/hello_qt.lpy
#+end_src
